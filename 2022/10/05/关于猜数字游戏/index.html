初学c语言，做了一个基于C语言的比较经典的猜数字游戏。
以下为源代码以及部分运行图片。

#include<stdio.h>
#include<time.h>
#include<stdlib.h>

int main()
{
    int min = 1;
    int x, y, z, max;
    int a = 0;
    
    printf("欢迎来到猜数字游戏，本游戏由云湍编写，版本号v1.5.5\n
    本游戏会随机生成一个数字，你需要用尽量少的次数猜出这个数字\n");
    printf("请选择模式：\n
    1.简单模式 数字范围1~100\n
    2.普通模式 数字范围1~1000\n
    3.困难模式 数字范围1~10000\n
    4.自定义模式\n输入对应的序号数字即可选择模式\n");
    scanf_s("%d", &z);

    switch (z)
    {
    case 1:
        max = 100;
        break;
    case 2:
        max = 1000;
        break;
    case 3:
        max = 10000;
        break;
    case 4:
        printf("请输入数字范围的上限\n");
        scanf_s("%d", &max);
        break;
    }
    
    srand(time(0));
    x = rand() % (max - min) + min;  // “%”为模运算
    
    while (1) 
    {
        printf("请输入一个数字\n");
        scanf_s("%d", &y);
        if (y < x)
            printf("小了\n");
        else if (y == x)
        {
            printf("恭喜，答对了\n");
            a = a + 1;
            break;
        }
        else if (y > x)
            printf("大了\n");
        a++;
    }
    
    printf("您已累计尝试%d次\n", a);

    switch (z)
    {
    case 1:
        if (a == 1)
        {
            printf("一发入魂！您的运气也太好了吧\n");
        }
        if (a > 1 && a < 6) 
        {
            printf("出神入化！您真是实力与运气并存\n");
        }
        if (a > 5 && a < 9)
        {
        printf("非常不错！您的水平已经很高了\n");
        }
        if (a > 8)
        {
        printf("勉勉强强！您的水平还需要提高\n");
        }
        break;
    
    case 2:
        if (a == 1)
        {
            printf("一发入魂！您的运气也太好了吧\n");
        }
        if (a > 1 && a < 16)
        {
            printf("出神入化！您真是实力与运气并存\n");
        }
        if (a > 15 && a < 19)
        {
            printf("非常不错！您的水平已经很高了\n");
        }
        if (a > 18)
        {
            printf("勉勉强强！您的水平还需要提高\n");
        }
        break;
    
    case 3:
        if (a == 1)
        {
            printf("一发入魂！您的运气也太好了吧\n");
        }
        if (a > 1 && a < 26)
        {
            printf("出神入化！您真是实力与运气并存\n");
        }
        if (a > 25 && a < 29)
        {
            printf("非常不错！您的水平已经很高了\n");
        }
        if (a > 28)
        {
            printf("勉勉强强！您的水平还需要提高\n");
        }
        break;
    
    case 4:
        printf("感谢您的参与，该模式下无评分系统\n");
        break;
    }

    system("pause");
    return 0;
}
